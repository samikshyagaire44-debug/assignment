JavaScript Basics

Adding interactivity and dynamic behavior to web pages

Why JavaScript is Important

JavaScript is a versatile, high-level programming language that brings web pages to life by adding interactivity, dynamic content updates, and complex functionality. While HTML provides structure and CSS handles styling, JavaScript is the programming language that makes websites interactive and responsive to user actions.

JavaScript runs directly in web browsers, making it the only programming language that can execute client-side code without requiring plugins or additional software. This unique capability allows developers to create rich, interactive user experiences including form validation, dynamic content updates, animations, games, and complex web applications.

Modern JavaScript has evolved far beyond simple webpage interactions. It's now used for server-side development (Node.js), mobile app development, desktop applications, and even machine learning. JavaScript's versatility and widespread adoption make it one of the most important programming languages for web developers to master.

How JavaScript Adds Interactivity

  • Event Handling: Responding to user clicks, keyboard input, mouse movements
  • DOM Manipulation: Dynamically changing HTML content and structure
  • Form Validation: Checking user input before form submission
  • Animations: Creating smooth transitions and visual effects
  • AJAX Requests: Loading data from servers without page refreshes
  • Local Storage: Saving user preferences and data in the browser
  • Real-time Updates: Updating content based on user actions or external data

Form Validation with JavaScript

Client-Side Form Validation
function validateForm() {
    // Get form elements
    const name = document.getElementById('user-name').value.trim();
    const email = document.getElementById('user-email').value.trim();
    const age = document.getElementById('user-age').value;
    const message = document.getElementById('user-message').value.trim();
    
    // Validation flags
    let isValid = true;
    let errorMessages = [];
    
    // Name validation
    if (name === '') {
        errorMessages.push('Name is required');
        isValid = false;
    } else if (name.length < 2) {
        errorMessages.push('Name must be at least 2 characters long');
        isValid = false;
    }
    
    // Email validation
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (email === '') {
        errorMessages.push('Email is required');
        isValid = false;
    } else if (!emailRegex.test(email)) {
        errorMessages.push('Please enter a valid email address');
        isValid = false;
    }
    
    // Age validation
    if (age === '') {
        errorMessages.push('Age is required');
        isValid = false;
    } else if (age < 13 || age > 120) {
        errorMessages.push('Age must be between 13 and 120');
        isValid = false;
    }
    
    // Message validation
    if (message === '') {
        errorMessages.push('Message is required');
        isValid = false;
    } else if (message.length < 10) {
        errorMessages.push('Message must be at least 10 characters long');
        isValid = false;
    }
    
    // Display results
    const errorDiv = document.getElementById('form-errors');
    const successDiv = document.getElementById('form-success');
    
    if (isValid) {
        errorDiv.style.display = 'none';
        successDiv.style.display = 'block';
        successDiv.innerHTML = '<p style="color: #28a745;">✓ Form submitted successfully!</p>';
        
        // Reset form after success
        setTimeout(() => {
            document.getElementById('validation-form').reset();
            successDiv.style.display = 'none';
        }, 3000);
    } else {
        successDiv.style.display = 'none';
        errorDiv.style.display = 'block';
        errorDiv.innerHTML = '<ul style="color: #dc3545;">' + 
            errorMessages.map(msg => '<li>' + msg + '</li>').join('') + 
            '</ul>';
    }
    
    return false; // Prevent actual form submission
}

Asynchronous JavaScript

Promises and Async/Await
// Promise example
function fetchUserData(userId) {
    return new Promise((resolve, reject) => {
        // Simulate API call
        setTimeout(() => {
            if (userId > 0) {
                resolve({
                    id: userId,
                    name: 'John Doe',
                    email: 'john@example.com'
                });
            } else {
                reject(new Error('Invalid user ID'));
            }
        }, 1000);
    });
}

// Using promises with .then()
fetchUserData(1)
    .then(user => {
        console.log('User data:', user);
        return user.name;
    })
    .then(name => {
        console.log('User name:', name);
    })
    .catch(error => {
        console.error('Error:', error.message);
    });

// Using async/await
async function getUserInfo(userId) {
    try {
        const user = await fetchUserData(userId);
        console.log('User info:', user);
        return user;
    } catch (error) {
        console.error('Failed to get user info:', error.message);
        return null;
    }
}

// Fetch API example
async function loadDataFromAPI() {
    try {
        const response = await fetch('https://jsonplaceholder.typicode.com/posts/1');
        const data = await response.json();
        console.log('API data:', data);
        return data;
    } catch (error) {
        console.error('API request failed:', error);
        return null;
    }
}

Local Storage

Saving Data in Browser
// Save data to localStorage
function saveUserPreferences() {
    const preferences = {
        theme: 'dark',
        language: 'en',
        notifications: true,
        lastLogin: new Date().toISOString()
    };
    
    localStorage.setItem('userPreferences', JSON.stringify(preferences));
    console.log('Preferences saved!');
}

// Load data from localStorage
function loadUserPreferences() {
    const saved = localStorage.getItem('userPreferences');
    
    if (saved) {
        const preferences = JSON.parse(saved);
        console.log('Loaded preferences:', preferences);
        return preferences;
    } else {
        console.log('No saved preferences found');
        return null;
    }
}

// Remove data from localStorage
function clearUserPreferences() {
    localStorage.removeItem('userPreferences');
    console.log('Preferences cleared!');
}

// Check if localStorage is supported
function isLocalStorageSupported() {
    try {
        const test = 'test';
        localStorage.setItem(test, test);
        localStorage.removeItem(test);
        return true;
    } catch (e) {
        return false;
    }
}

JavaScript Best Practices

  • Use meaningful variable names: Choose descriptive names that explain the purpose
  • Declare variables properly: Use const for constants, let for variables
  • Handle errors gracefully: Use try-catch blocks for error handling
  • Validate user input: Always validate and sanitize user input
  • Use strict mode: Add 'use strict'; at the beginning of scripts
  • Avoid global variables: Use modules and proper scoping
  • Comment your code: Explain complex logic and algorithms
  • Use modern JavaScript features: Leverage ES6+ features for cleaner code
  • Test your code: Write unit tests and test across different browsers
  • Optimize performance: Minimize DOM manipulation and use efficient algorithms
